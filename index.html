Samples

Sample 1


// Creates 320 × 320 px raphael canvas
var r = Raphael("holder", 320, 320);

// Create a gauge with 0 min angle and 360 max angle
var g = r.gauge(0, 360);

// Attach Background to gauge with x=140 and y=140 center point
g.bg(r.circle(140, 140, 140).attr({fill:"#ccc"}), [140, 140]);

// Attach Pointer to gauge with x=140 and y=5 center point
g.pointer(r.rect(0, 0, 140, 10).attr({fill: "#000"}), [140, 5]);

// Move pointer with random value
setInterval(function(){
	var percent = Math.floor(Math.random()*100);
	g.move(percent);
}, 2000);

Sample 2


// Creates 200 × 134 px raphael canvas
var r2 = Raphael("holder2", 200, 134);

// Create a gauge with 17 min angle and 163 max angle
var g2 = r2.gauge(17, 163);

// Attach Background to gauge with x=100 and y=94 center point
g2.bg(r2.image("gauge-bg.png", 0, 0, 200, 134), [100, 94])

// Attach Pointer to gauge with x=67 and y=13 center point
g2.pointer(r2.image("gauge-pointer.png", 0, 0, 84, 27), [67, 13]);

// Move pointer to 50% without animation
g2.move(50, false);

// Move pointer with random value
setInterval(function(){
	var percent = Math.floor(Math.random()*100);
	g2.move(percent);
}, 1500);
